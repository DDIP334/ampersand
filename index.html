                        int main() {
                            int i,j;
                            char a[15],f[15];
                            printf("Enter 15 characters without giving space:");
                            for(i=0; i<15; i++) {
                                scanf("%c,",&a[i]);
                                f[i]=1;
                                printf("\nEntered characters are: %c",a[i]);
                        
                            }
                            for(i=0; i<15; i++) {
                                if(f[i]==0)
                                    continue;
                                for(j=i+1; j<15; j++) {
                                    if(f[j]==0)
                                        continue;
                                    if(a[i]==a[j]) {
                                        f[i]++;
                                        f[j]=0;
                                    }
                                }
                            }
                            for(i=0; i<15; i++) {
                                if(f[i]!=0) {
                                    printf("\nFrequency of %c is %d",a[i],f[i]);
                                }
                            }
                            return 0;
                        }

using namespace cv;
using namespace std;

int main() {
    // 1. Open the webcam
    VideoCapture cam(0); // 0 = default camera
    if (!cam.isOpened()) {
        cout << " Could not open the camera!" << endl;
        return -1;
    }

    Mat frame;
    cout << "Show your product to the camera.\n";
    cout << "Press 's' to capture the image.\n";

    // 2. Show webcam feed and wait for 's' key to capture
    while (true) {
        cam >> frame; // Read frame from camera
        imshow("Webcam View", frame); // Show frame
        if (waitKey(1) == 's') {
            imwrite("captured/product.jpg", frame); // Save image
            break;
        }
    }

    cam.release(); // Close the webcam
    destroyAllWindows();

    // 3. Load the captured image and a template image to compare
    Mat captured = imread("captured/product.jpg"); // From webcam
    Mat templateImg = imread("templates/chemistry.jpg");
    // Known product

    if (captured.empty() || templateImg.empty()) {
        cout << "Failed to load images!" << endl;
        return -1;
    }

    // 4. Convert both images to grayscale for easier comparison
    cvtColor(captured, captured, COLOR_BGR2GRAY);
    cvtColor(templateImg, templateImg, COLOR_BGR2GRAY);

    // 5. Resize both to same size
    resize(captured, captured, Size(300, 300));
    resize(templateImg, templateImg, Size(300, 300));

    // 6. Compare images using template matching
    Mat result;
    matchTemplate(captured, templateImg, result, TM_SQDIFF); // TM_SQDIFF = lower is better

    double minVal;
    minMaxLoc(result, &minVal); // Find lowest matching value

    cout << "Matching Score: " << minVal << endl;

    // 7. Show result based on score
    if (minVal < 1e9) {
        cout << "\n=====  BILL =====" << endl;
        cout << "Product: Chemistry Book" << endl;
        cout << "Price: Rs. 500" << endl;
        cout << "===================" << endl;
    }
    else {
        cout << "Product not recognized." << endl;
    }

    return 0;
}
